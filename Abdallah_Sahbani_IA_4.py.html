# Ask user to input the their monthly spending budget
# input:Gets user input as a string ,float Converts the string input to a floating-point number.
# budget:Stores the numeric value of the user's budget for later use
# main() allows for program entry
def main():
    budget = float(input("Enter the amount you have budgeted for the month: $"))

    # budget_difference(): This is the function call
    total_expenses = budget_difference()

    # orders the function to convert the status of the budget to an output
    budget_status(budget, total_expenses)


# def defines the function then calculates the budget difference
def budget_difference():
    # Initialize the  collector for total expenses
    total_expenses = 0

    # Initialize the expense variable with any non-zero value to engage the loop
    expense = -1  # Any non-zero value to engage the loop

    # Start the loop for entering expenses
    while expense != 0:
        # Prompt the user for the next expense then format float
        expense = float(input("Enter an expense (enter 0 to stop): $"))

        # If adds the expense to the total if it's not zero
        if expense != 0:
            total_expenses += expense

    # Return the total expenses gives the output after formulating all expenses
    return total_expenses


# Define the function to display the budget status
def budget_status(budget, total_expenses):
    # Calculate the difference between the budget and total expenses
    difference = budget - total_expenses
     # The ':,.2f' ensures the value is displayed with commas
    # If the difference is positive, it means the user is under budget
    if difference > 0:
        print(f"You are under budget by ${difference:,.2f}. Well done!")
    # If the difference is negative, it means the user is over budget
    elif difference < 0:
        print(f"You are over budget by ${abs(difference):,.2f}. Control your impulsive spending!")
    # If the difference is zero, the user is exactly on budget
    else:
        print(f"You are on budget. Good planning!")
# Call the main function to  restart the program
main()
